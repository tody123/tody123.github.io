安装node

clearup

rm -rf $HOME/.nvm
sed -i ‘/nvm/d’ $HOME/.bashrc

clone nvm

cd $HOME
git clone .nvm
cd .nvm
git checkout ‘git describe –abbrev=0 –tags’

activate nvm

echo “export NVM_NODEJS_ORG_MIRROR=http://npm.taobao.org/mirrors/node“ >> $HOME/.bashrc
echo “. $HOME/.nvm/nvm.sh” >> $HOME/.bashrc
source $HOME/.bashrc

#config
nvm install v6.2.0
nvm alias default v6.2.0

退出node执行环境

process.exit()；
